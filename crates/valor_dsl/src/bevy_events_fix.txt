    /// Mouse position.
    pub position: (f32, f32),
}

impl bevy::ecs::event::EntityEvent for OnClick {
    fn target(&self) -> bevy::prelude::Entity {
        self.entity
    }

    fn event_target_mut(&mut self) -> &mut bevy::prelude::Entity {
        &mut self.entity
    }
}
